from django.db import models

class SMS(models.Model):
    phone_number = models.CharField(max_length=20)
    response_code = models.CharField(max_length=20)
    message = models.TextField()
    created_at = models.DateTimeField(auto_now_add=True)




from django.shortcuts import render
from .models import SMS

def send_sms(request):
    if request.method == 'POST':
        phone_number = request.POST.get('phone_number')
        message = request.POST.get('message')
        # send the SMS and get the response code
        response_code = send_sms(phone_number, message)
        # save the SMS details to the database
        sms = SMS(phone_number=phone_number, message=message, response_code=response_code)
        sms.save()
        return render(request, 'sms_sent.html')
    else:
        return render(request, 'send_sms.html')

import arkesel

# Set up Arkesel client
client = arkesel.Client('your-api-key')

def send_sms(recipients, message):
    # Send SMS to multiple recipients
    for recipient in recipients:
        response = client.sms.send(
            message=message,
            sender='YourSenderName',
            recipients=[recipient]
        )
        print(response)
